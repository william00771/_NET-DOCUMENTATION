		import * as signalR from "@microsoft/signalr";

		let connection = new signalR.HubConnectionBuilder()
		    .withUrl("/hubs/view")
		    .build();

		connection.start().then(startSuccess, startFail);

			.send("hubMethod", param1, param2) - Send to hub
				.then((val) => ...)
