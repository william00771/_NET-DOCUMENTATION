	calculations
		=SUMMA(A1:A10) - SUM
			=SUMMAOM(A1:120,"<10") - SUMIF - Calculate sum based on if statement
			=SUMMAOMS(A1:120,F2:F11,"=Male", H2:H11, ">50") - SUMIFS - calculate sum based on multible ifs
		=OM(A1>10; "Value if true"; "Value if false") - ELSE/IF
		=LETARAD(search_value; A1:B10; 2; FALSKT) - VLOOKUP, ex lookup names inside one table to another table and return salary of those specific employees
			=LETARD(A2, 'TableName'B:J, ColumnToReturnFromTableLookupex9)
		=TRIM(A1) - Removes all space between
		=RIGHT/LEFT(4) - Returns number of characters from right or left - Probably many expressions that work here
		=FIND("a", "Monday", 1) - Returns index of a inside monday starting from index 1
		=LEN(A2) - Number of charachters
		=REPLACE(A2,1,5,"Mark") - what to replace, where to start, number of characters, what
		=ANTAL(C1:C10) - COUNT
			=ANTALOM(A1:A20,"<12") - COUNTIF
			=ANTALOMS() - COUNTIFS
		=MIN(A1:A10) - MIN
		=MAX(B1:B10) - MAX
		=MEDEL(B1:B10) - AVERAGE
		=HÄMTA(B2:D10; 2; 3) - INDEX
		=PASSA("Värde"; A1:A10; 0) - MATCH
		=OMFEL(A1/B1; "Fel") - IFERROR
